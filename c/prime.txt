```plaintext
To check whether a given number is prime or not.

#include <stdio.h>

int main(){
    int num,i,flag=0;
    printf("\n enter a number");
    scanf("%d",&num);
    if (num <= 1) {
        printf("\n the number is not prime");
        return 0;
    }
    for(i=2;i<=num/2;i++){
        if(num%i==0){
            flag=1;
            break;
        }
    }
    if(!flag)
        printf("\n the number is prime");
    else
        printf("\n the number is not prime");

    return 0;
}
```
