To implement a program to find sum and average of no's in an array.

#include<stdio.h>
int main()
{
int n,i,sum=0;
float avg;
printf("Enter the number of elements \n");
scanf("%d", &n);
int arr[n];
printf("Enter %d numbers \n", n);
for(i=0;i<n;i++)
{
scanf("%d", &arr[i]);
sum+=arr[i];
}
avg=(float)sum/n;
printf("Sum is %d \n", sum);
printf("Average is %f", avg);
return 0;
}
