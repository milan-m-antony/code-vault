To implement a program to check whether a number is Armstrong or Not.

#include <stdio.h>
#include <math.h>

int main() {
    int num, original, rem, digits = 0;
    int sum = 0;

    printf("Enter a number: ");
    scanf("%d", &num);

    original = num;

    while (num != 0) {
        digits++;
        num = num / 10;
    }

    num = original;
    while (num != 0) {
        rem = num % 10;
        sum = sum + (int)pow(rem, digits);
        num = num / 10;
    }

    if (sum == original)
        printf("The number is an Armstrong number");
    else
        printf("The number is not an Armstrong number");

    return 0;
}
