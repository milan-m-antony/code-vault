To implement a menu driven program to perform various Array operations- Insertion, Deletion, Traversal.

#include<stdio.h>
#include<stdlib.h>
void main()
{
int a[50];
int x,i,j,p,s,n;
printf("Enter the size of the array: \n ");
scanf("%d",&s);
printf("Enter the elements of the array: \n ");
for(i=0;i<s;i++)
{
scanf("%d",&a[i]);
}
while(1)
{
printf("\nEnter the array operation: \n 1.Insertion \n 2.Deletion \n 3.Traversal \n 4.Exit \n");
scanf("%d",&n);
switch(n)
{
case 1:
printf("Enter the number to insert: \n");
scanf("%d", &x);
printf("Enter the position to insert: \n");
scanf("%d", &p);
for (i = s; i > p-1; i--) {
a[i] = a[i - 1];
}
a[p-1] = x;
s++;
printf("Updated array: \n");
for (i = 0; i < s; i++)
printf("%d ", a[i]);
break;
case 2:
printf("Enter the number to delete: \n");
scanf("%d",&x);
for(i=0;i<s;i++)
{
if(a[i]==x)
{
for(j=i;j<s-1;j++)
{
a[j]=a[j+1];
}
s--;
i--;
}
}
printf("Updated array: \n");
for(i=0;i<s;i++)
printf("%d ",a[i]);
break;
case 3:
printf("Array elements: \n");
for(i=0;i<s;i++)
printf("%d ",a[i]);
break;
case 4:
exit(0);
default:
printf("\n invalid choice");
}
}
}
