To implement a menu driven program to perform stack operations using array.

#include<stdio.h>
#include<stdlib.h>
#define SIZE 100
int stack[SIZE];
int top=-1;
int i;
void push()
{
int value;
if(top==SIZE-1)
{
printf("\n Stack overflow! Cannot push more elements \n");
}
else
{
printf("Enter the value to push");
scanf("%d",&value);
stack[++top]=value;
printf("%d pushed to stack \n",value);
}
}
void pop()
{
if(top==-1)
{
printf("\n Stack underflow! No elements to pop \n");
}
else
{
printf("Popped element %d \n",stack[top--]);
}
}
void peek()
{
if(top==-1)
{
printf("\n Stack is empty \n");
}
else
{
printf("Top element is %d \n",stack[top]);
}
}
void display()
{
if(top==-1)
{
printf("\n Stack is empty\n");
}
else
{
printf("\n Stack elements\n");
for(i=top;i>=0;i--)
{
printf("%d \n",stack[i]);
}
}
}
int main()
{
int choice;
while(1)
{
printf("\n---STACK MENU---\n");
printf("1. Push\n 2. Pop\n 3. Peek\n 4. Display\n 5. Exit\n Enter your choice\n");
scanf("%d",&choice);
switch(choice)
{
case 1:
push();
break;
case 2:
pop();
break;
case 3:
peek();
break;
case 4:
display();
break;
case 5:
exit(0);
break;
default:
printf("Invalid choice");
}
}
return 0;
}
