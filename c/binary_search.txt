To implement a program to perform Binary search.

#include<stdio.h>
int main()
{
int n,x,low,high,mid,i,found;
printf("Enter the number of elements\n");
scanf("%d",&n);
int arr[n];
printf("Enter %d sorted elements\n",n);
for(i=0;i<n;i++)
{
printf("Element %d\n",i+1);
scanf("%d",&arr[i]);
}
printf("Enter the element to search\n");
scanf("%d",&x);
low=0;
high=n-1;
found=0;
while(low<=high)
{
mid=(low+high)/2;
if(arr[mid]==x)
{
printf("Element found at index %d\n",mid);
found=1;
break;
}
else if(arr[mid]<x)
{
low=mid+1;
}
else
{
high=mid-1;
}
}
if(!found)
{
printf("Element not found in the array\n");
}
return 0;
}
