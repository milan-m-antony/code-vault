<?php

$con = mysqli_connect("localhost", "root", "", "milan");
if (!$con) 
echo "Database not connected: ";


if (isset($_GET['id'])) {
    $id = intval($_GET['id']);

    $sql = "DELETE FROM student WHERE id = $id";
    if (mysqli_query($con, $sql)) {
        header("Location: view.php");
        exit();
    } else {
        echo "Error deleting record: " . mysqli_error($con);
    }
} else {
    echo "No ID provided.";
}
?>
